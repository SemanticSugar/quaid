 31 Q.Textbox = Class.extend('Textbox', /** @lends Q.Textbox# */{
 32     
 33     /**
 34      * @class <p>Simple textbox class. Right now just provides a function to get the box's selection</p>
 35      *
 36      * @augments Q.Class
 37      * @param box the textbox
 38      * @param options your config
 39      * @constructs
 40      **/
 41     init: function(box, options){
 42         this._super(box, options, {});
 43         this.box = $(box);
 44     },
 45     
 46     /** <p>Pass through to jQuery's input.val() method</p> */
 47     val: function(value){
 48         return this.box.val(value);
 49     },
 50     
 51     /** <p>Gets the currently selected text in the box.</p> */
 52     getSelection: function(){
 53         
 54         var textComponent = this.box[0];
 55         var selectedText = '';
 56         
 57         if (document.selection != undefined)
 58         {
 59             textComponent.focus();
 60             var sel = document.selection.createRange();
 61             selectedText = sel.text;
 62         }
 63         // Mozilla version
 64         else if (textComponent.selectionStart != undefined)
 65         {
 66             var startPos = textComponent.selectionStart;
 67             var endPos = textComponent.selectionEnd;
 68             selectedText = textComponent.value.substring(startPos, endPos)
 69         }
 70         
 71         return selectedText;
 72 
 73     }

 76 Q.IntBox = Q.Textbox.extend('IntBox', /** @lends Q.IntBox# */{
 77     /**
 78      *  @class <p>Int box. Only allows integers to be typed into the box.</p>
 79      *
 80      *  <pre class="code">
 81      *  $('input#my-textbox').IntBox();
 82      *  </pre>
 83      *
 84      *  @augments Q.Textbox
 85      *  @param box the textbox
 86      *  @param options your config
 87      *  @constructs
 88      **/
 89     init: function(box, options){
 90         
 91         var defs = {
 92             onFail: function(key){ return false; },
 93             maxDigits: 7
 94         };
 95         
 96         var settings = $.extend({}, this.defaults, options);
 97         
 98         this._super(box, settings);
 99         
100         self = this;
101         
102         this.box.keypress(function (e){
103             
104             //check for max number of digits. But still let them type control chars,
105             //and integers if they have selected something (resulting in an overwrite)
106             if(self.val().length > settings.maxDigits-1 && !(e.which in Q.controlChars) && !self.getSelection())
107                 return false;
108             
109             //if the letter is not digit then display error and don't type anything
110             if( (e.which!=8 && e.which!=0 && e.which!=13 && (e.which<48 || e.which>57)))
111                 return settings.onFail.call(self, e.which);
112             
113             return true;
114         });
115         
116         return this;
117     },
118     
119     /**
120      * <p>Default options.</p>
121      *
122      * <pre class="code">
123      * onFail: function(keyCode){ return false; },
124      * maxDigits: 7
125      * </pre>
126      */
127     defaults: {
128         onFail: function(key){ return false; },
129         maxDigits: 7
130     }

167 Q.Form = Class.extend('Form', /** @lends Q.Form */{
168         //static members
169         /**
170          * <p>Default options.</p>
171          * <pre class="code">
172          * // enforces data types for form elements. use 'input_name':'type'
173          * // supported types: int, decimal, + whatever you put in Q.Form.dataTypes
174          * dataTypes: {
175          *     //i.e
176          *     //date_input: 'date'
177          * },
178          * validationOptions: {},
179          * defaultData: {},
180          * resetInitially: false,
181          * onLoad: function(data){},
182          * onSubmit: function(form){}
183          * </pre>
184          */
185         defaults: {
186             dataTypes: {},
187             validationOptions: {},
188             defaultData: {},
189             resetInitially: false,
190             
191             onLoad: function(data){},
192             onSubmit: function(form, formData){}
193         },
194         
195         /**
196          * Data types that can be specified for an element in the Q.Form class. These will setup the
197          * element for the type specified. i.e. an int element will only allow the user to type integers.
198          * A hypothetical date element would hookup a date picker, etc. Feel free to add your own.
199          **/
200         dataTypes: {
201             'int': function(elem){
202                 elem.IntBox();
203             },
204             'decimal': function(elem){
205                 elem.DecimalBox();
206             }
207         }
208     }, /** @lends Q.Form# */{
209     /**
210      * @class <p>Handles validation via validate plugin. Will load default data. Can force types on
211      * elements with the dataType option. Also provides comvenience methods like
212      * load() (from a dict), reset(), getElement etc.
213      * </p>
214      *
215      * @augments Q.Class
216      * @constructs
217      **/
218     init: function(container, options){
219         var self = this;
220         self._super(container, $.extend(true, {}, self._class.defaults, options));
221         settings = self.settings;
222         
223         if(container.is('form'))
224             self.form = container;
225         else{
226             self.form = container.find('form');
227             if(!self.form.length){
228                 $.error('Cannot find form in', container, '!! You need a form. Fo\' Real!');
229                 return;
230             }
231         }
232         
233         self.form.validate(settings.validationOptions);
234         
235         if(settings.resetInitially)
236             this.reset();
237         
238         for(var k in settings.dataTypes){
239             var elem = self.getElement(k);
240             if(!elem.length)
241                 continue;
242             
243             self._setElementDataFilter(elem, settings.dataTypes[k]);
244         }
245         
246         self.form.submit(function(){
247             return self._onSubmit(arguments);
248         });
249     },
250     
251     /**
252      * <p>Apply the data types filters. i.e. make some text boxen IntBoxes, or DecimalBoxes, etc.</p>
253      * @private
254      */
255     _setElementDataFilter: function(elem, type){
256         if($.isFunction(this._class.dataTypes[type])){
257             this._class.dataTypes[type].call(this, elem);
258         }
259     },
260     
261     /**
262      * <p>Set data in an element. Dont use this. use val()</p>
263      * @private
264      */
265     _setData: function(elem, data){
266         if(elem.attr('type') == 'radio'){
267             elem.each(function(){
268                 if($(this).val() == data) $(this).attr('checked', 'checked');
269             });
270         }
271         else if(elem.attr('type') == 'checkbox'){
272             if(data) elem.attr('checked', 'checked');
273             else elem.attr('checked', null);
274         }
275         else
276             elem.val(data);
277     },
278     
279     /**
280      * <p>The form's submit event hook. You can return false to </p>
281      * @private
282      */
283     _onSubmit: function(){
284         if($.isFunction(this.settings.onSubmit)){
285             var formData = $.parseUrlParams(this.form.formSerialize());
286             return this.settings.onSubmit.call(this, this.form, formData, arguments);
287         }
288         return true;
289     },
290     
291     /**
292      * <p>Will return the HTML element with the name specified if possible.</p>
293      * @returns an HTML element corresponding to the passed in name
294      */
295     getElement: function(name){
296         return this.form.find('input[name="'+name+'"], textarea[name="'+name+'"], select[name="'+name+'"]');
297     },
298     
299     /**
300      * <p>Works like jQuery val() function. If the val parameter is not specified, val()
301      * will return the value of the element corresponding to the passed in name. If val param
302      * is specified, it will set the HTML element's value to val. This properly handles
303      * radio buttons and checkboxen. Set radio buttons by specifying the desired radio's
304      * value string. Set checkboxen with a bool.</p>
305      *
306      * @param name the name of the HTML element to value get/set
307      * @param val (optional) the value to set. Will set if specified, will get if not.
308      * @returns undefined if val specified, the value of the HTML element if not.
309      */
310     val: function(name, val){
311         var elem = this.getElement(name);
312         
313         //set data?
314         if(val != undefined){
315             return this._setData(elem, val);
316         }
317         
318         //get data
319         if(elem.eq(0).attr('type') == 'radio'){
320             return elem.filter(':checked').val();
321         }
322         else if(elem.attr('type') == 'checkbox'){
323             return elem.is(':checked');
324         }
325         return elem.val();
326     },
327     
328     /**
329      * <p>Set focus on the first text element in the form.</p>
330      */
331     focusFirst: function(){
332         this.form.find('input[type="text"]:first, textarea:first').focus();
333     },
334     
335     /**
336      * <p>Will reset the form based on the settings.defaultData dict. If an element's name
337      * is not in the dict, the value will be cleared/unchecked</p>
338      */
339     reset: function(){
340         var elems = this.form.find('input, textarea, select');
341         var defs = this.settings.defaultData;
342         
343         for(var i = 0; i < elems.length; i++){
344             var elem = elems.eq(i);
345             var n = elem.attr('name');
346             if(elem.attr('type') != 'submit'){
347                 var data = '';
348                 if(n in defs)
349                     data = defs[n];
350                 this._setData(elem, data);
351             }
352         }
353     },
354     
355     /**
356      * <p>Loads data into the form. data is in the format:</p>
357      * <pre class="code">{
358      *   'element_name': 'value',
359      *   'checkbox': true,
360      *   ....
361      * }</pre>
362      *
363      * @param data the data object to load into the form
364      **/
365     load: function(data){
366         if($.isFunction(this.settings.onLoad)) this.settings.onLoad.call(this, data);
367         
368         for(var k in data){
369             this.val(k, data[k]);
370         }
371     },
372     
373     /**
374      * <p>Will hide the element which Q.Form was called on. </p>
375      */
376     hide: function(){
377         this.container.hide();
378     },
379     
380     /**
381      * <p>Will show the element which Q.Form was called on. </p>
382      */
383     show: function(){
384         this.container.show();
385         this.focusFirst();
386     },
387     
388     /**
389      * <p>Will hide the submit the HTML form.</p>
390      */
391     submit: function(){
392         this.form.submit();
393     }

396 Q.AsyncForm = Q.Form.extend('AsyncForm', /** @lends Q.AsyncForm */{
397         /**
398          * <p>Default options. Extends {@link Q.Form} defaults.</p>
399          * <pre class="code">
400          * loaderLocation: {position: 'absolute', bottom: 5, left: 5}, //passed into Q.Loader
401          * ajaxOptions: {dataType: 'json'}, //passed into $.ajax()
402          * validationOptions: {}, //passed into the validation plugin
403          * onSuccess: function(data){},
404          * onFail: function(err_type, errors){}
405          * </pre>
406          */
407         defaults: {
408             loaderLocation: {position: 'absolute', bottom: 5, left: 5},
409             ajaxOptions: {dataType: 'json'},
410             autoGenValidationOptions: false,
411             validationOptions: {},
412             onSuccess: function(data){},
413             onFail: function(errorType, errors){}
414         }
415     },/** @lends Q.AsyncForm# */{
416     /**
417      * @class <p>
418      * Extends Form to allow for async form submission.
419      * Provides validation via the bassistance validate plugin.
420      * Provides async form submission via malsup's form plugin.
421      * Provides a loading image inside the form via this library's Q.Loader object.
422      * </p>
423      *
424      * @augments Q.Form
425      * @constructs
426      **/
427     init: function(container, settings){
428         var self = this;
429         settings = $.extend(true, {}, self._class.defaults, settings);
430         
431         if(settings.autoGenValidationOptions){
432             var rules = settings.validationOptions.rules || {};
433             var inputs = container.find('input, select, textarea');
434             
435             for(var i=0; i < inputs.length; i++)
436                 if(!(inputs[i].name in rules))
437                     rules[inputs[i].name] = {required: false};
438             settings.validationOptions.rules = rules;
439         }
440         
441         //hook the async form submission to the validation plugin
442         settings.validationOptions.submitHandler = function(validForm){
443             self.loader.startLoading();
444             settings.ajaxOptions.form = self.form;
445             var opts = $.extend(settings.ajaxOptions, {
446                 success: function(data){
447                     self.onSuccess.apply(self, arguments);
448                     self.loader.stopLoading();
449                 },
450                 applicationError: function(){
451                     self.onFail.apply(self, arguments);
452                     self.loader.stopLoading();
453                 }
454             });
455             self.form.ajaxSubmit(opts);
456         };
457         
458         self._super(container, settings);
459         
460         self.loader = self.form.Loader({location: self.settings.loaderLocation});
461     },
462     
463     onSuccess: function(){
464         if($.isFunction(this.settings.onSuccess))
465             this.settings.onSuccess.apply(this, arguments);
466     },
467     
468     onFail: function(){
469         if($.isFunction(this.settings.onFail))
470             this.settings.onFail.apply(this, arguments);
471     }
472 });
