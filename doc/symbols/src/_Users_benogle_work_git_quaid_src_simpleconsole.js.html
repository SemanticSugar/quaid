 29 var Q = $[$.QUAID];
 30 
 31 $.SimpleConsole = {
 32     cookieOpts: { path: '/', expires: 10 },
 33     messageMarkup: '<div class="console-message"></div>',
 34     markup: '<div id="console">' +
 35                 '<a id="console-expand-link" href="#">[-]</a>' +
 36                 '<h3 id="console-head">Console</h3>' +
 37                 '<div id="console-body"></div>' +
 38             '</div>'
 39 };
 40 
 41 Q.SimpleConsole = Class.extend('SimpleConsole', {
 42     init: function(container, settings){
 43         this.console = null;
 44         var console = $('#console');
 45         if(console.length > 0)
 46             this.console = console;
 47         this._super(container, settings);
 48     },
 49     
 50     _makeConsoleIfNecessary: function(){
 51         if(this.console)
 52             return this.console;
 53         
 54         console = $($.SimpleConsole.markup);
 55         console.hide();
 56         $('body').append(console);
 57         this.console = console;
 58         
 59         this.body = console.find('#console-body');
 60         this.toggle = console.find('#console-expand-link');
 61         
 62         $this = this;
 63         this.toggle.click(function(){
 64             var left = parseInt($this.console.css('left'));
 65             $this.changeConsoleDisplay(left < 0);
 66             return false;
 67         });
 68         
 69         if($.cookie){
 70             var remember = $.cookie('console');
 71             this.changeConsoleDisplay(remember == 'show');
 72         }
 73         
 74         return console;
 75     },
 76     
 77     _humanify: function(thing){
 78         try{
 79             if(typeof(thing) == "string")
 80                 return "'" + thing + "'";
 81             else if($.isFunction(thing)){
 82                 var fnstr = (thing+'');
 83                 return fnstr.substring(0, fnstr.indexOf('{'));
 84             }
 85             else if($.isArray(thing)){
 86                 //for(var i = 0; i < thing.length; i++)
 87                 //    thing[i] = this._humanify(thing[i]);
 88                 return '[' + thing.join(', ') + ']';
 89             }
 90             else if(thing instanceof Object){
 91                 var pair = null;
 92                 for( var x in thing ){
 93                     pair = [x, thing[x]];
 94                 }
 95                 
 96                 var str = '{';
 97                 if(pair && pair.length == 2)
 98                     str += pair[0] + ': ' + pair[1]/*this._humanify(pair[1])*/ + ' ... ';
 99                 str += '}';
100                 
101                 return str;
102             }
103             return thing;
104         }
105         catch(err){
106             return ''+thing;
107         }
108     },
109     
110     _getOutput: function(thing){
111         var str = '';
112         
113         if($.isArray(thing)){
114             str = this._humanify(thing);
115         }
116         else if(thing instanceof Object){
117             
118             str = 'Object: {<br>';
119             
120             for(k in thing){
121                 
122                 data = thing[k];
123                 
124                 str += '  ' + k + ': ' + this._humanify(data) + ',<br>';
125             }
126             
127             str += '}<br>';
128         }
129         else
130             str += thing;
131         
132         return str;
133     },
134     
135     _setConsoleCookie: function(val){
136         if($.cookie)
137             $.cookie('console', val, $.SimpleConsole.cookieOpts);
138     },
139     
140     changeConsoleDisplay: function (doShow){
141         var consoleWid = this.console.width();
142         if(doShow){
143             this._setConsoleCookie('show');
144             this.toggle.text('[-]');
145             this.console.animate({left: 0}, 200);
146         }
147         else{
148             this._setConsoleCookie('hide');
149             this.toggle.text('[+]');
150             this.console.animate({left: 32 - consoleWid}, 200);
151         }
152     },
153     
154     log: function(){
155         
156         this._makeConsoleIfNecessary();
157         
158         var data = '';
159         
160         for(var i = 0; i < arguments.length; i++)
161             data += this._getOutput(arguments[i]) + ' ';
162         
163         data = $($.SimpleConsole.messageMarkup).html(data);
164         if(this.body){
165             this.body.append(data);
166             this.body.animate({scrollTop: 99999}, 20);
167         }
168         this.console.show();
169         
170     }
171 });
