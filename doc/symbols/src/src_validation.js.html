 32 /**
 33  * <p>Monkey patch for the jquery ajax function. It handles our json
 34  * conventions. It handles 500 errors. It makes you coffee. It entertains your
 35  * girlfriend so you can get some shit done.</p>
 36  *
 37  * <p>This patch is a thin layer that does some error handling and introduces a
 38  * new option to the options object:</p>
 39  *
 40  * <pre class="code">applicationError()</pre>
 41  *
 42  * <p>So when everything is good, the server will return json in the form:</p>
 43  *
 44  * <pre class="code">{ status: 'success', otherStuff: ... }</pre>
 45  *
 46  * <p>When there is an application error it will return something like:</p>
 47  *
 48  * <pre class="code">{
 49  *     status: 'fail',
 50  *     errors: [
 51  *         {value: 'hi@example.com', message: 'Email address in use', field: 'email_address'},
 52  *         {code: 64, message: 'Ads step is incomplete'}
 53  *     ]
 54  * }</pre>
 55  *
 56  * <p>In this case your options.applicationError(errortype, errors) callback will
 57  * be called. errortype will be 'applicationerror' and errors will be split into
 58  * general and field type errors:</p>
 59  *
 60  * <pre class="code">{
 61  *     general: [
 62  *         {code: 64, message: 'Ads step is incomplete'}
 63  *     ],
 64  *     field: [
 65  *         {value: 'hi@example.com', message: 'Email address in use', field: 'email_address'}
 66  *     ]
 67  * }</pre>
 68  *
 69  * <p>Field errors will be passed automatically through {@link Q.asyncErrors.add}
 70  * and will get shoved into your form if possible.</p>
 71  **/
 72 $.ajax = function( options ) {
 73 
 74     //If json, we put our functions in the options dict for success and error.
 75     //also define a new callback in options called applicationError.
 76     if(options.dataType && options.dataType == 'json'){
 77     
 78         //options now accepts a new parameter: applicationError
 79     
 80         //errortype: 'servererror' - internal server error; errors -  http code (500, 501, etc.)
 81         //           'applicationerror' - means errors from our system; errors - {general: [], field: []}
 82         function defErrorCb(errortype, errors){};
 83         if( !$.isFunction(options.applicationError) )
 84             options.applicationError = defErrorCb;
 85         
 86         function _handleAppErrors(data){
 87             var errors = {
 88                 general: [],
 89                 field: []
 90             };
 91             
 92             //split the errors into the different types. Then the client
 93             //will have an easier time parsing through the errors.
 94             for(var i = 0; data.errors && i < data.errors.length; i++){
 95                 if('field' in data.errors[i] && data.errors[i].field)
 96                     errors.field.push(data.errors[i]);
 97                 else
 98                     errors.general.push(data.errors[i]);
 99             }
100             
101             if(data.errors.length > 0)
102                 Q.asyncErrors.add(data.errors, options.form);
103             
104             options.applicationError.call(this, 'applicationerror', errors);
105             
106             var uh = Q.asyncErrors.getUnhandledErrors();
107             if(uh && $.isFunction(Q.handleApplicationErrors))
108                 Q.handleApplicationErrors.call(this, uh);
109         }
110         
111         //So success gets anything that is a 200. However, we may have application
112         //errors coming in (data.status == 'fail'). So we will extract those
113         //responses and call the error callback.
114         var originalSuccess = options.success || function(){};
115         function successHandler(data, status, xhr){
116             
117             //aborted request causes success to be called with empty data. Boo.
118             if(!xhr.status) return;
119             
120             if(!data)
121                 $.warn('No data returned from the server! Request args:', options);
122             
123             Q.handleSuccess(data, options);
124             
125             // fail if they have a 'fail' in the status or they specify errors.
126             if( data && (data.status == 'fail' || (data.errors && data.errors.length)))
127                 _handleAppErrors(data);
128             // just pass through for success....
129             else
130                 originalSuccess.call(this, data, status);
131             
132         }//end success handler
133         options.success = successHandler;
134         
135         var originalError = options.error || function(){};
136         function errorHandler(xhr, status, errorThrown){
137             
138             if(xhr.status >= 500){
139                 var handled = false;
140                 var data = null;
141                 try {
142                     data = jQuery.httpData( xhr, options.dataType, options );
143                 } catch(err) {
144                     $.warn('ERROR: Failed to parse error data from the server. Is your response in', options.dataType, '?', 'Error:', err);
145                 }
146                 
147                 options.applicationError.call(this, 'servererror', xhr.status);
148                 Q.handleServerError.call(this, data, xhr, status, errorThrown);
149             }
150             else if(xhr.status >= 400){
151                 try {
152                     var data = jQuery.httpData( xhr, options.dataType, options );
153                     _handleAppErrors(data);
154                 } catch(err) {
155                     options.applicationError.call(this, 'parsererror', err);
156                 }
157             }
158             else{
159                 var original = '[Unknown]';
160                 try{ original = xhr.responseText || xhr.responseXML; }
161                 catch(err){}
162                 
163                 var error = '[Unknown Error]';
164                 if(errorThrown)
165                     error = errorThrown.number + ': ' + errorThrown.description;
166                 
167                 $.log('$.ajax Error ', error, ', http status: ', xhr.status, ':', status, '; original', original);
168             }
169             
170             originalError.call(this, xhr, status, errorThrown);
171             
172         }//end error handler
173         options.error = errorHandler;
174         
175     }//end json monkey patchage
176     
177     return $.ajaxOriginal(options);
178 };

193 /**
194  * On any Application errors. When the server returns a 400, or a status: 'fail' from a
195  * 200 request.
196  **/
197 Q.handleApplicationErrors = function(errors){
198     for(var i = 0; i < errors.length; i ++)
199         $.warn('Unhandled async application error "', errors[i].message ,'" :', errors[i]);
200     Q.asyncErrors.handle(errors);
201 };

203 /**
204  * @namespace
205  * 
206  * <p>Allows us to push validation errors from the
207  * server through the jquery validation framework. Here's how it works.</p>
208  *
209  * <p>You setup some validation rules:</p>
210  *
211  * <pre class="code">var rules = {
212  *     rules: { something: 'required' },
213  *     messages: { something: 'omg you need something' }
214  * };
215  * $('form#my-form').validate(rules);</pre>
216  *
217  * <p>$.fn.validate will call {@link Q.getValidationOptions} which will add the 'asyncError' validation method to your
218  * rules. You can add these asyncError validation methods manually if you like:</p>
219  *
220  * <pre class="code">var rules = {
221  *     rules: { something: {
222  *         required: true,
223  *         asyncError: false //can be true or false. false will disable it.
224  *     }},
225  *     messages: { something: 'omg you need something' }
226  * };</pre>
227  * 
228  * <p>You make a post to some action on the server which will return some errors,
229  * then Quaid's monkey-patched {@link $.ajax} function adds the errors to Q.asyncErrors. Q.asyncErrors
230  * will display the errors just like regular client side errors.</p>
231  *
232  * <pre class="code">$.post('/myaction', {param: 'wow'}, function(data){
233  *     if(data.status == 'fail')
234  *         Q.asyncErrors(data.errors, $('form#my-form'));
235  * }, 'json');</pre>
236  *
237  * <p>The data from above request failure should look something like:</p>
238  *
239  * <pre class="code">data = {
240  *     status: 'fail',
241  *     errors: [
242  *         {field: 'name', value: '', message: 'Name is too short!'},
243  *         {field: 'email', value: 'wow@ok', message: 'Email is not valid!'}
244  *     ]
245  * }</pre>
246  **/
247 Q.asyncErrors = {
248     
249     fieldErrors: {},
250     errors: [],
251     
252     /**
253      * <p>Add a list of errors to this Q.asyncErrors. The form will be revalidated and your
254      * errors will be pushed through the client side validation framework.</p>
255      *
256      * @param errors A list of error dictionaries. Should be in the format:
257      * <pre class="code">[
258      *     {field: 'name', value: '', message: 'Name is too short!'},
259      *     {field: 'email', value: 'wow@ok', message: 'Email is not valid!'}
260      * ]</pre>
261      */
262     add: function(errors, form){
263       
264         Q.asyncErrors.fieldErrors = {};
265         if(errors)
266             Q.asyncErrors.errors = Q.asyncErrors.errors.concat(errors);
267         
268         if(errors && $.fn.valid){
269             //iterate through the fields, and save the errors in our special
270             //little errors construct. This construct is what the validation
271             //rule will read.
272             for(var i = 0; i < errors.length; i++){
273                 if(! errors[i].field ) continue;
274                 
275                 var field;
276                 if(form)
277                     field = form.find('[name="'+ errors[i].field +'"]');
278                 else
279                     field = $('[name="'+ errors[i].field +'"]');
280 				if(field.length && field.is(":visible")){
281 					//Infer the form from the field. I hope like hell I am not
282 					//dumb enough to have fields from multiple forms coming back in
283 					//in this errors dict. 
284 					if(!form || form.length == 0)
285 						form = field.parents('form');
286 					
287 					//we save the old value so the rule knows what value caused the error.
288 					//if the value changes, then the field will be valid.
289 					Q.asyncErrors.fieldErrors[errors[i].field] = {
290 						message: errors[i].message,
291 						val: field.val(),
292                         original: errors[i]
293 					};
294 				}
295             }
296             
297             //force form validation so the framework will read our new errors.
298             if(form){
299                 //there may be multiple forms we need to attend to...
300                 form.each(function(){
301                     $(this).valid();
302                 });
303             }
304         }
305     },
306     
307     handle: function(error){
308         if(!$.isArray(error))
309             error = [error];
310         for(var i = 0; i < error.length; i++)
311             error[i]._handled = true;
312     },
313 
314     /**
315      * <p>Gets all the unhandled errors.</p>
316      */
317     getUnhandledErrors: function(){
318         var unhandled = [];
319         
320         for(var i = 0; i < Q.asyncErrors.errors.length; i++)
321             if(!Q.asyncErrors.errors[i]._handled)
322                 unhandled.push(Q.asyncErrors.errors[i]);
323         
324         return unhandled;
325     },
326     
327     clear: function(){
328         Q.asyncErrors.errors = [];
329         Q.asyncErrors.fieldErrors = {};
330     }
331 };

334 if($.validator && $.fn.validate){
335     /**
336      * New validation method. Works with Q.asyncErrors to display the
337      * errors returned from the server through the validation framework.
338      *
339      * Allows for connection to a second field. Specified like this:
340      *
341      * rules:{
342      *     field1: {asyncError: {connect: 'field2', display: true} },
343      *     field1: {asyncError: {connect: 'field1', display: false} }
344      * }
345      *
346      * The 'display: true|false' is important in this case. The error for field1
347      * will be displayed. Field1's value will be assumed to be
348      * field1.value()+field2.value(). When the value in field2 is changed it will
349      * just trigger a check on field1's value. 
350      **/
351     $.validator.addMethod(
352         "asyncError",
353         
354         //check fn
355         function(value, element, params) {
356             //'this' is the validator
357             
358             //if they specify asyncError: false, do nothing.
359             if(!params) return true;
360             
361             //assemble the element's current value...
362             var cur = $(element).val();
363             
364             //..and the value that caused the error.
365             var err = '';
366             if(element.name in Q.asyncErrors.fieldErrors)
367                 err = Q.asyncErrors.fieldErrors[element.name].val || '';
368             
369             // if this element is connected to another, and it is the display element,
370             // concat the connected element's current value and error value.
371             if(params instanceof Object && params.connect && params.display){
372                 cur += $('[name="'+ params.connect +'"]').val();
373                 if(params.connect in Q.asyncErrors.fieldErrors)
374                     err += Q.asyncErrors.fieldErrors[params.connect].val || '';
375             }
376             
377             //if it is connected, but not the display element, just trigger a check
378             //on the main display element
379             else if(params instanceof Object && params.connect && !params.display){
380                 this.check($('[name="'+ params.connect +'"]'));
381                 return true;
382             }
383             
384             //basically: do we have an error for this element?
385             //has the element's value NOT changed since we got the error?
386             //if both true, fail.
387             fail = (element.name in Q.asyncErrors.fieldErrors && cur == err);
388             
389             return /*this.optional(element) || */ !fail;
390         },
391         
392         //the message fn; extract the message we got from the server
393         function(params, element){
394             //'this' is the validator
395             
396             var error = Q.asyncErrors.fieldErrors[element.name];
397             
398             var message = error.message;
399             
400             Q.asyncErrors.handle(error.original);
401             
402             return message;
403         }
404     );

406     /**
407      * <p>Overall default validation options. You probably want to override this for, say,
408      * your own error label placement.</p>
409      */
410     Q.defaultValidationOptions = {
411         //insert your own defaults here!
412     };
413     
414     /**
415      * <p>Overrides {@link Q.defaultValidationOptions} with newopts, normalizes all your rules,
416      * and adds 'asyncError' validation methods to all rules where it is not already specified.</p>
417      *
418      * <p>Called automatically by the new validate() function.</p>
419      */
420     Q.getValidationOptions = function(newopts){
421         
422         function normalizeRule(data) {
423             if( typeof data == "string" ) {
424                 var transformed = {};
425                 $.each(data.split(/\s/), function() {
426                     transformed[this] = true;
427                 });
428                 data = transformed;
429             }
430             return data;
431         }
432         
433         var opts = $.extend({}, Q.defaultValidationOptions, newopts);
434         
435         if(opts.rules){
436             //force asyncError rule into each and every field. 
437             for(k in opts.rules){
438                 opts.rules[k] = normalizeRule(opts.rules[k]);
439                 
440                 if( !('asyncError' in opts.rules[k]) )
441                     opts.rules[k].asyncError = true;
442             }
443         }
444         
445         //$.log(opts);
446         return opts;
447     };
448     
449     //monkey patch the validation plugin
450     $.fn._oldValidate = $.fn.validate;
451     $.fn.validate = function(options){
452         var opts = Q.getValidationOptions(options);
453         return $(this)._oldValidate(opts);
454     };
