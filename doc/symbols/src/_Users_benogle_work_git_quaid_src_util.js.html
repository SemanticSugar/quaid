 72         for( var i = 0; i < max; i++ )

120     /**
121      * <p>Sort of string interpolation.</p>
122      * <pre class="code">
123      * replace('/blah/{0}/apsd/{1}', ['wow', 'omg']); // returns '/blah/wow/apsd/omg'
124      * replace('/blah/{bar}/apsd/{foo}', {foo: 'wow', bar: 'omg'}); // returns '/blah/omg/apsd/wow'
125      * </pre>
126      * @param str the string to replace
127      * @param vars the vars to put into the string in either list or dict form
128      * @returns a str with the variables contained in vars replaced
129      **/
130     replace: function(str, vars){
131         
132         //turns out the js replace fn only replaces the first str. I am not using regexs. Obvi.
133         function repl(s, replace, withThis){
134             while(s.indexOf(replace) > -1)
135                 s = s.replace(replace, withThis);
136             return s;
137         }
138         
139         if($.isArray(vars)){
140             for( var i = 0; i < vars.length; i++)
141                 str = repl(str, '{' + i + '}', vars[i])
142         }
143         else if($.isObject(vars)){
144             for( var k in vars )
145                 str = repl(str, '{' + k + '}', vars[k])
146         }
147         return str;
148     },

150     /**
151      * <p>Will commify the number. 10000 becomes 10,000.</p>
152      * 
153      * @param num the number
154      * @param roundDigits (optional) will make commifyNumber round the integer. i.e.
155      *   12345 with roundDigits == 2 would net 12,300
156      * @returns a commified number
157      **/
158     commifyNumber: function(num, roundDigits){
159        
160         //will round the int if possible
161         if(roundDigits){
162             var factor = Math.pow(10, roundDigits);
163             num = parseInt(Math.round(Math.round(num / factor) * factor));
164         }
165         
166         num = ''+num;
167         var vals = num.split('.');
168         var whole = vals[0];
169         
170         var part = null;
171         if(vals.length > 1)
172             part = vals[1];
173         
174         var neg = num[0] == '-';
175         whole = whole.replace('-', '');
176         
177         var result = '';
178         
179         whole = whole.split('').reverse();
180        
181         for(var i = 0; i < whole.length; i++){
182             if( i != 0 && i%3 == 0 )
183                 result = ',' + result;
184             result = whole[i] + result;
185         }
186        
187         if(part)
188             result += '.'+part;
189        
190         if(neg)
191             result = '-' + result;
192         
193         return result;
194     },

196     /**
197      * <p>Will parse a string date into a Date object.</p>
198      * <p>From www.eyecon.ro date picker under MIT. Thanks Stefan Petre.</p>
199      * 
200      * @param date the string date
201      * @param format
202      * @returns a Date object
203      **/
204     parseDate: function(date, format){
205         if (!date) return null;
206         if (date.constructor == Date || $.isNumber(date))
207             return new Date(date);
208         if (!$.isString(date) || !format) return null;
209         
210         var parts = date.split(/\W+/);
211         var against = format.split(/\W+/), d, m, y, h, min, now = new Date();
212         for (var i = 0; i < parts.length; i++) {
213             switch (against[i]) {
214                 case 'd':
215                 case 'e':
216                     d = parseInt(parts[i],10);
217                     break;
218                 case 'm':
219                     m = parseInt(parts[i], 10)-1;
220                     break;
221                 case 'Y':
222                 case 'y':
223                     y = parseInt(parts[i], 10);
224                     y += y > 100 ? 0 : (y < 29 ? 2000 : 1900);
225                     break;
226                 case 'H':
227                 case 'I':
228                 case 'k':
229                 case 'l':
230                     h = parseInt(parts[i], 10);
231                     break;
232                 case 'P':
233                 case 'p':
234                     if (/pm/i.test(parts[i]) && h < 12) {
235                         h += 12;
236                     } else if (/am/i.test(parts[i]) && h >= 12) {
237                         h -= 12;
238                     }
239                     break;
240                 case 'M':
241                     min = parseInt(parts[i], 10);
242                     break;
243             }
244         }
245         return new Date(
246             y === undefined ? now.getFullYear() : y,
247             m === undefined ? now.getMonth() : m,
248             d === undefined ? now.getDate() : d,
249             h === undefined ? 0 : h,
250             min === undefined ? 0 : min,
251             0
252         );
253     },

255     /**
256      * <p>Will format a Date object into a string.</p>
257      * <p>From www.eyecon.ro date picker under MIT. Thanks Stefan Petre.</p>
258      * 
259      * @param date the Date object
260      * @param format
261      * @returns a string date
262      **/
263     formatDate: function(date, format) {
264         var m = date.getMonth();
265         var d = date.getDate();
266         var y = date.getFullYear();
267         var w = date.getDay();
268         var s = {};
269         var hr = date.getHours();
270         var pm = (hr >= 12);
271         var ir = (pm) ? (hr - 12) : hr;
272         if (ir == 0) ir = 12;
273         var min = date.getMinutes();
274         var sec = date.getSeconds();
275         var parts = format.split(''), part;
276         for ( var i = 0; i < parts.length; i++ ) {
277             part = parts[i];
278             switch (parts[i]) {
279                 case 'a': //day name short
280                     part = Q.daysShort[w];
281                     break;
282                 case 'A': //day name long
283                     part = Q.days[w];
284                     break;
285                 case 'b': //month name short
286                     part = Q.monthsShort[m];
287                     break;
288                 case 'B': //month name long
289                     part = Q.months[m];
290                     break;
291                 case 'C': // century
292                     part = 1 + Math.floor(y / 100);
293                     break;
294                 case 'd': //zero padded day
295                     part = $.zeroPad(d, 2);
296                     break;
297                 case 'e': // non zero padded day
298                     part = d;
299                     break;
300                 case 'H': // zero padded 24-hour hour
301                     part = $.zeroPad(hr, 2);
302                     break;
303                 case 'I': // zero padded 12 hour hour
304                     part = $.zeroPad(ir, 2);
305                     break;
306                 case 'j': //zero padded day of the year
307                     var now = new Date(date.getFullYear(), date.getMonth(), date.getDate(), 0, 0, 0);
308 					var then = new Date(date.getFullYear(), 0, 0, 0, 0, 0);
309 					var time = now - then;
310 					var dy = Math.floor(time / (24*60*60*1000));
311                     part = $.zeroPad(dy, 3);
312                     break;
313                 case 'k': // non zero padded 24-hour hour
314                     part = hr;
315                     break;
316                 case 'l': // non zero padded 12-hour hour
317                     part = ir;
318                     break;
319                 case 'm': //zero padded month 
320                     part = $.zeroPad((1+m), 2);
321                     break;
322                 case 'M': //zero padded min
323                     part = $.zeroPad(min, 2);
324                     break;
325                 case 'p': //am pm lower
326                     part = pm ? "pm" : "am";
327                     break;
328                 case 'P': //am pm upper
329                     part = pm ? "PM" : "AM";
330                     break;
331                 case 's': //seconds from epoch
332                     part = Math.floor(date.getTime() / 1000);
333                     break;
334                 case 'S': // seconds zero padded
335                     part = $.zeroPad(sec, 2);
336                     break;
337                 case 'u': // day of week index + 1
338                     part = w + 1;
339                     break;
340                 case 'w': // day of week number (index)
341                     part = w;
342                     break;
343                 case 'y': // year (2 digit)
344                     part = ('' + y).substr(2, 2);
345                     break;
346                 case 'Y': // 4 digit year
347                     part = y;
348                     break;
349             }
350             parts[i] = part;
351         }
352         return parts.join('');
353     },

359     /**
360      * <p>Takes a url, with or without params in string form, and a params dict.

362      * 
363      * @param wholeUrl the entire url (or just the query string)

365      * @returns a new URL

367     extendUrl: function(wholeUrl, params){
368         var paramStart = wholeUrl.indexOf('?');
369         var url = wholeUrl;
370         var oldParams = '';
371         if(paramStart > -1){
372             url = wholeUrl.substring(0, paramStart);
373             oldParams = wholeUrl.substring(paramStart);
374         }
375         
376         return url + '?' + $.serializeUrlParams($.extend($.parseUrlParams(oldParams), params));
377     },

389     /**
390      * <p>Parses out the params from a query string. Deserialization.</p>
391      * 
392      * @param q the query string (or whole url) from which you wish to extract params.
393      * @returns a object containing the url's params
394      **/
395     parseUrlParams: function(q){
396         if(!q) return {};
397         var paramStart = q.indexOf('?');
398         if(paramStart > -1)
399             q = q.substring(paramStart+1);
400         
401         var reg = /(?:^|&)([^&=]*)=?([^&]*)/g;
402         
403         var dataDict = {};
404         
405         q.replace( reg, function ( match, key, val ) {
406             if (match)
407                 dataDict[key] = val;
408         });
409         
410         return dataDict;
411     },

452 /**
453  * @namespace
454  * <p>Functions for formatting data into something a human might like to look at.</p>
455  * <p>These are useful to invoke on data that comes back from the server.</p>

458     
459     /** <p>The parse format for the date() and datetime() formatter.</p> */
460 	dateParseFormat: 'm.d.Y H:M',
461     /** <p>The output format for the datetime() formatter.</p> */
462     datetimeFormat: 'b e, Y H:M',

466     /**
467      * <p>Will parse a string like 'decimal(3)' into a dict like {fn:'decimal', num:3}.
468      * This is used by the get() function. This format is useful for plugins that need to
469      * use the data formatters as an option. i.e. lets pretend you are writing a table
470      * plugin. The plugin user could specify formatters for each column like so:</p>
471      *
472      * <pre class="code">
473      * var tableOptions = {
474      *     columnFormats: {
475      *         num_column: 'number',
476      *         other_column: 'percent(4)',
477      *         decimal_column: 'decimal(2)',
478      *     }
479      * };
480      * </pre>
481      *
482      * @param interp a string like 'decimal(2)'
483      * @returns an object like {fn:'decimal', num:3}
484      */
485     _parseInterpreter: function(interp){
486         var re = /([A-Za-z_]+)(\(([0-9]+)\))?/;
487         var res = {};
488         var r = re.exec(interp);
489         res.fn = r[1];
490         res.num = r[3] ? parseInt(r[3]) : null;
491         return res;
492     },

494     /**
495      * <p>Takes a string and a raw value. Will call _parseInterpreter() on the string and call
496      * the resulting formatter with the passed in raw value as a parameter. i.e.</p>
497      *
498      * <pre class="code">
499      * var DF = Q.DataFormatters;
500      * DF.get('decimal(2)', 1.234545); //will call DF.decimal(1.234545, 2); resulting in '1.23'
501      * DF.get('number', 12345); //will call DF.number(12345); resulting in '12,345'
502      * </pre>
503      *
504      * @param fn a string containing a name of one of the formatters i.e. 'number'
505      * @param val the raw value to format
506      *
507      * @returns the result of the formatter call if the formatter can be found.
508      * The raw value if the formatter cannot be found.
509      */
510     get: function(fn, val){
511         var DI = Q.DataFormatters;
512         var res = DI._parseInterpreter(fn);
513         
514         if(res.fn in DI){
515             var params = [val];
516             if(res.num) params.push(res.num);
517             
518             return DI[res.fn].apply(this, params);
519         }
520         return val;
521     },
