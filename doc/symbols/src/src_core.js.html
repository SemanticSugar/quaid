 38 var fnTest = /xyz/.test(function(){xyz;}) ? /\b(_super|_class)\b/ : /.*/;

 44 /**

 52 (/** @lends Q.Class# */function(){
 53     
 54     var initializing = false;
 55 
 56     // The base Class implementation
 57     // add Class to window (this is window)
 58     this.Class = function(){
 59     };
 60     /**
 61      * <p>The base class constructor. Provides setting extension. Puts the settings in
 62      * this.settings. Puts the container element in this.container. This constructor takes
 63      * two settings params so in your subclass you can call:</p>
 64      *
 65      * <pre class="code">this._super(container, userSettings, newClassDefaults);</pre>
 66      *
 67      * @class <p>This is the base class all your new classes will inherit from. Check out the
 68      * main usage: {@link Q.Class.extend}.</p>
 69      * 
 70      * @constructs
 71      * @param container The jQuery object that this class relates to
 72      * @param settings The user's passed in settings
 73      * @param defaults Your subclass' defaults
 74      * @name Q.Class
 75      */
 76     this.Class.prototype.init = function(container, settings, defaults){
 77         this.container = container;
 78         this.settings = $.extend({}, defaults, settings);
 79     };
 80  
 81     /**
 82      * <p>Create a new class that inherits from Class.</p>
 83      * 
 84      * <p>newClassName is optional. When specified, Quaid will put your class under
 85      * Q[newClassName] and will wrap it in a jQuery plugin at $.fn[newClassName]. For example:</p>
 86      *
 87      * <pre class="code">
 88      * Class.extend('Hidden',{
 89      *     init: function(container, settings){
 90      *         this._super(container, settings, {});
 91      *         this.container.hide();
 92      *     }
 93      * });
 94      * var div = $('#some-div');
 95      * var hidden = div.Hidden({setting: 'blah'});
 96      * //or
 97      * var hidden = new Q.Hidden(div, {setting: 'blah'});
 98      * </pre>
 99      *
100      * <p>If newClassName is not specified the new class will not be put in the Q or
101      * the $.fn namespace. Example:</p>
102      *
103      * <pre class="code">
104      * var MyNewClass = Class.extend({
105      *     init: function(blah, wow, ok){
106      *         this.omg = wow;
107      *     }
108      * });
109      *
110      * var yay = new MyNewClass(2, 'derp', 'foo');
111      * $.log(yay.omg);
112      * </pre>
113      *
114      * @function 
115      * @name Q.Class.extend
116      * @param newClassName (optional) The name of your subclass.
117      * @param staticProperties (optional) static properties of your new class. Accessable via Q.YourClass.$param
118      * @param properties The protoype object for your new class
119      * @returns Your shiny new class!
120      */
121     Class.extend = function(/*newClassName, staticMembers, prop*/) {
122         
123         function isStr(s){return typeof s == "string" || s instanceof String;}
124         function isObj(o){return o instanceof Object;}
125         
126         // bogle modifications:
127         //  * allow for auto creation of a node method.
128         //  * allow ability to specify static members
129         var newClassName, staticMembers, prop;
130         
131         if(arguments.length == 1 && isObj(arguments[0]))
132             prop = arguments[0]; //just prototype extension
133         else if(arguments.length == 2 && isObj(arguments[0]) && isObj(arguments[1]))
134             staticMembers = arguments[0], prop = arguments[1];
135         else if(arguments.length == 2 && isStr(arguments[0]) && isObj(arguments[1]))
136             newClassName = arguments[0], prop = arguments[1];
137         else if(arguments.length == 3 && isStr(arguments[0]) && isObj(arguments[1]) && isObj(arguments[2]))
138             newClassName = arguments[0], staticMembers = arguments[1], prop = arguments[2];
139         else
140             throw new Error('extend() not called correctly!');
141         
142         if(newClassName){
143             
144             if($.fn[newClassName]){
145                 throw Error("You cannot define '"+newClassName+"' more than once!");
146             }
147             
148             //setup a node/element method for this new class. Maybe kind of magic.
149             $.fn[newClassName] = function(options){ 
150                 var elems = this;
151                 var ret = [];
152                 elems.each(function(){
153                     var elem = $(this);
154                     var obj = elem.data(newClassName);
155                     if(! obj){
156                         if(! $[$.CLASS_NAMESPACE][newClassName])
157                             $.log('Cannot find $.' + $.CLASS_NAMESPACE + '.' + newClassName + '!');
158                         else{
159                             obj = new $[$.CLASS_NAMESPACE][newClassName](elem, options);
160                             elem.data(newClassName, obj);
161                             obj.TYPE = newClassName;
162                         }
163                     }
164                     ret.push(obj);
165                 });
166                 if(ret.length == 1) return ret[0];
167                 if(!ret.length) return null;
168                 return ret;
169             };
170         }
171         // end bogle mod
172         
173         var _super = this.prototype;
174    
175         // Instantiate a base class (but only create the instance,
176         // don't run the init constructor)
177         initializing = true;
178         var prototype = new this();
179         initializing = false;
180        
181         // Copy the properties over onto the new prototype
182         for (var name in prop) {
183             // Check if we're overwriting an existing function
184             prototype[name] = typeof prop[name] == "function" &&
185                 typeof _super[name] == "function" && fnTest.test(prop[name]) ?
186                 (function(name, fn, cls){
187                     return function() {
188                         var tmp = this._super;
189                         var tmpCls = this._class;
190                        
191                         // Add a new ._super() method that is the same method
192                         // but on the super-class
193                         this._super = _super[name];
194                         this._class = cls;
195                        
196                         // The method only need to be bound temporarily, so we
197                         // remove it when we're done executing
198                         var ret = fn.apply(this, arguments);       
199                         this._super = tmp;
200                         this._class = tmpCls;
201                        
202                         return ret;
203                     };
204                 })(name, prop[name], Class) :
205                 prop[name];
206         }
207        
208         // The dummy class constructor
209         function Class() {
210             // All construction is actually done in the init method
211             if ( !initializing && this.init )
212                 this.init.apply(this, arguments);
213             //this._class = Class;
214         }
215         
216         if(staticMembers)
217             $.extend(Class, staticMembers);
218         
219         // Populate our constructed prototype object
220         Class.prototype = prototype;
221        
222         // Enforce the constructor to be what we expect
223         Class.constructor = Class;
224         
225         // And make this class extendable
226         Class.extend = arguments.callee;
227         
228         if(newClassName)
229             //save the new class in $.class[newClassName]
230             $[$.CLASS_NAMESPACE][newClassName] = Class;
231         
232         return Class;
233     };
234 })();
