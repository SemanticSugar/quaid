 31 Q.Loader = Class.extend('Loader',/** @lends Q.Loader */{
 32         /**
 33          * <p>Default options.</p>
 34          * <pre class="code">
 35          * image: '16x16_arrows.gif', //path to your loading gif
 36          * location: {position: 'absolute', right: 5, top: 5}, //css opts; can be 'center'
 37          * opacity: 0.4 //opacity of the container while loading
 38          * </pre>
 39          */
 40         defaults: {
 41             image: '16x16_arrows.gif',
 42             location: {position: 'absolute', right: 5, top: 5},
 43             opacity: 0.4 //opacity of the container while loading     
 44         }
 45     },/** @lends Q.Loader# */{
 46     /**
 47      * @class
 48      * <p>An object that will display a loading gif. Creates
 49      * its own markup. Called on the container within which you would like to see
 50      * the loader.</p>
 51      *
 52      * <p>It will handle multiple requests. You can call startLoading() multiple times
 53      * before calling stopLoading(). startLoading() effectively increments a counter
 54      * and stopLoading() decrement the counter. The image will be shown when the counter
 55      * is non-zero and hidden when zero.</p>
 56      * 
 57      * <p>This is useful if more than one link can be clicked and start parallel
 58      * requests that need the same loader.</p>
 59      *
 60      * <p>Note that your container probably has got to have position set to something
 61      * like relative. If it is a position: static the plugin will set it to relative so
 62      * you aren't confused when it looks like the image isn't showing up then emailing me
 63      * saying it doesn't work.</p>
 64      * 
 65      * @example
 66      *
 67      * var myLoader = $('#my-container').Loader();
 68      *
 69      * $.post('/blah', function(){
 70      *     myLoader.stopLoading();
 71      * });
 72      * myLoader.startLoading();
 73      *
 74      * $.post('/blah-again', function(){
 75      *     myLoader.stopLoading();
 76      * });
 77      * myLoader.startLoading();
 78      *
 79      * @augments Q.Class
 80      * @param container The jQuery object
 81      * @param options Your config.
 82      * @constructs
 83      **/
 84     init: function(container, options){
 85         
 86         this._super(container, options, Q.Loader.defaults);//this._class.defaults);
 87         
 88         //startLoading counts up, stopLoading counts
 89         //down. When count > 0, loader shown; when == 0, loader hides.
 90         this.loadingCount = 0;
 91         
 92         this.loaderCreated = false;
 93     },
 94     
 95     /**
 96      * Creates the image markup and jams it in the container element.
 97      * @private
 98      */
 99     _createLoader: function(){
100         this.loaderCreated = true;
101         
102         var self = this;
103         var set = this.settings;
104         var container = $('<div/>');
105         
106         var image = $('<img/>');
107         image[0].src = this.settings.image;
108         
109         container.append(image);
110         container.addClass('inline-loader');
111         container.hide();
112         
113         this.container.append(container);
114         
115         this.loader = container;
116         
117         //do this so the abs positioned image has a ref point.
118         if(this.container.css('position') == 'static')
119             this.container.css('position', 'relative')
120         
121         if(this.settings.location == 'center'){
122             
123             //the image might take a bit to load.
124             //When it loads we set the position...
125             var objImagePreloader = new Image();
126             objImagePreloader.onload = function() {
127                 
128                 var h = self.container.height();
129                 var w = self.container.width();
130                 self.setLocation({
131 					position: 'absolute',
132 					top: h/2 - objImagePreloader.height/2,
133                     left: w/2 - objImagePreloader.width/2
134 				});
135                 
136                 objImagePreloader.onload=function(){};
137             };
138             objImagePreloader.src = self.settings.image;
139         }
140         else
141             self.setLocation(self.settings.location);
142     },
143     
144     /**
145      * Shows the loading image
146      * @private
147      */
148     _show: function(){
149         if(!this.loaderCreated)
150             this._createLoader();
151         
152         if(!this.isLoading()){
153             this.oldOpacity = this.container.css('opacity');
154             this.container.css('opacity', '' + this.settings.opacity)
155             this.loader.css('display', 'block');
156         }
157     },
158     
159     /**
160      * Hides the loading image
161      * @private
162      */
163     _hide: function(){
164         if(this.isLoading()){
165             if(!this.oldOpacity || this.oldOpacity == '1')
166                 //remove opacity or opacity:1 will hose everything in IE. 
167                 this.container.css('opacity', null);
168             else
169                 this.container.css('opacity', this.oldOpacity);
170             this.loader.css('display', 'none');
171         }
172     },
173     
174     /**
175      * <p>Sets the location (really any css params) of the loading image. Can be used at any time.</p>
176      * 
177      * @param locCssObj An object that will be applied to the loadimg image via jQuery.css()
178      */
179     setLocation: function(locCssObj){
180         this.settings.location = locCssObj;
181         if(this.loader)
182             this.loader.css(locCssObj);
183     },
184     
185     /**
186      * <p>Lets you know if the loading image is loading or not.</p>
187      * 
188      * @returns a bool 
189      */
190     isLoading: function(){
191         if(this.loader)
192             return this.loader.css('display') == 'block';
193         return false;
194     },
195     
196     /**
197      * <p>Will increment the loading count and show the loader if not already shown.</p>
198      */
199     startLoading: function(){
200         this.loadingCount++;
201         
202         this._show();
203     },
204     
205     /**
206      * <p>Will decrement the loading count and potentially hide the loader.</p>
207      */
208     stopLoading: function(){
209         this.loadingCount = Math.max(0, this.loadingCount-1);
210         
211         if(this.loadingCount == 0)
212             this._hide();
213     }
214 });

216 Q.AsyncLoader = Q.Loader.extend('AsyncLoader', /** @lends Q.AsyncLoader */{
217         /**
218          * <p>Default options. Extends {@link Q.Loader} defaults.</p>
219          * <pre class="code">
220          * ajaxOptions: { //ajaxOptions passed into $.ajax
221          *     type: 'POST',
222          *     dataType: 'json',
223          *     onPostLoad: function(){}
224          * }
225          * </pre>
226          */
227         defaults: {
228             ajaxOptions: {
229                 type: 'POST',
230                 dataType: 'json'
231             },
232             onPostLoad: null
233         }
234     },/** @lends Q.AsyncLoader# */{
235     /**
236      * @class
237      * <p>Connects an async request to a loading image. Exposes a load() function
238      * that you can pass a url, params and callbacks to.</p>
239      *
240      * @augments Q.Loader
241      * @param container The jQuery object
242      * @param options Your config.
243      * @constructs
244      **/
245     init: function(container, options){
246         //must deep copy because of the ajaxOptions...
247         this._super(container, $.extend(true, {}, this._class.defaults, options));
248     },
249     
250     /**
251      * <p>Loads data from the server, shows/hides the loading image, calls your callbacks, makes you coffee.</p>
252      *
253      * @param url The url to make the request to
254      * @param params An object filled with params to the server
255      * @param successfn success callback. If not specified, will use the callback
256      * specified in the ajaxOptions. Signature function(data){}
257      * @param failfn fail callback. If not specified, will use the callback
258      * specified in the ajaxOptions. Signature: function(error_type, errors){}
259      */
260     load: function(url, params, successfn, failfn){
261         
262         if(!url) {
263             $.log('AsyncLoader.load on ', this.container, ': load() needs a url, f00! Seriously.');
264             return false;
265         }
266         
267         var self = this;
268         
269         self.startLoading();
270         
271         function _genHandler(str, fn){
272             return function(){
273                 var cb = fn
274                 
275                 if($.isFunction(cb))
276                     cb.apply(this, arguments);
277                 
278                 self.stopLoading();
279                 
280                 if($.isFunction(self.settings.onPostLoad))
281                     self.settings.onPostLoad.call(this, str, arguments);
282             };
283         }
284         
285         return $.ajax($.extend(self.settings.ajaxOptions, {
286             url: url,
287             data: params,
288             success: _genHandler('success', $.isFunction(successfn) ? successfn : self.settings.ajaxOptions.success),
289             applicationError: _genHandler('fail', $.isFunction(failfn) ? failfn : self.settings.ajaxOptions.applicationError)
290         }));
291     }

294 Q.SingleResourceAsyncLoader = Q.AsyncLoader.extend('SingleResourceAsyncLoader', /** @lends Q.SingleResourceAsyncLoader */{
295         /**
296          * <p>Default options. Extends {@link Q.AsyncLoader} defaults.</p>
297          * <pre class="code">
298          * url: '', //Your resource to load
299          * onSuccess: function(data){}, //when request is successful
300          * onFail: function(error_type, errors){}
301          * </pre>
302          */
303         defaults: {
304             url: '',
305             onSuccess: null,
306             onFail: null
307         }
308     }, /** @lends Q.SingleResourceAsyncLoader# */{
309     /**
310      * @class 
311      *
312      * <p>Will deal with race conditions. Extends the AsyncLoader.</p>
313      *
314      * <p>Basically, it provides a load() function. You call init with your url and
315      * your call backs. THen you call load() with your params. It will handle race conditions.
316      * This is super useful when youre loading data into some interface object that can only accept
317      * one set of data at a time, but the user can induce multiple requests.
318      * i.e. a dynamically loaded table, a dynamic graph, etc.</p>
319      *
320      * @augments Q.AsyncLoader
321      * @param container The jQuery object
322      * @param options Your config.
323      * @constructs
324      **/
325     init: function(container, options){
326         
327         var defs = {
328             url: '',
329             onSuccess: null,
330             onFail: null
331         };
332         this._super(container, $.extend({}, this._class.defaults, options));
333         
334         this.currentRequest = null;
335         this.lastParams = {};
336     },
337     
338     /**
339      * <p>Connects an async request to a loading image.</p>
340      *
341      * @param params An object filled with params to pass to the server
342      */
343     load: function(params){
344         this.lastParams = params || this.lastParams;
345         
346         var req = this._super(this.settings.url, this.lastParams, this.settings.onSuccess, this.settings.onFail);
347         
348         if(req && this.currentRequest){
349             this.currentRequest.abort();
350             this.stopLoading();
351         }
352         this.currentRequest = req;
353     }
354 });
